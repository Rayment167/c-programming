                                       Copy One Array to Another


#include <stdio.h>
int main() {
    int s[100], d[100];
    int n, i;
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &s[i]);
    }
    for (i = 0; i < n; i++) {
        d[i] = s[i];
    }
    printf("Copied array:\n");
    for (i = 0; i < n; i++) {
        printf("%d ", d[i]);
    }
    return 0;
}

output

Enter the number of elements: 5
Enter 5 elements:
1 2 3 4 5
Copied array:
1 2 3 4 5
