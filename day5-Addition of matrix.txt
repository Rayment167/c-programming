                                                Addition of matrix

#include <stdio.h>
int main() {
   int a[2][2]={{1,2},{3,4}};
   int b[2][2]={{5,6},{7,8}};
   int c[2][2]={{9,10},{11,12}};
   int d[2][2],i,j;
   for(i=0;i<2;i++) {
        for(j=0;j<2;j++)  {
            d[i][j]= a[i][j]+ b[i][j] + c[i][j];
            printf("%d\t",d[i][j]);
        }
        printf("\n");
   }
    return 0;
}

output

15	18	
21	24
