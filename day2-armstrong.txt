#include <stdio.h>
#include <math.h>
int main() {
    int num, o, r, n = 0;
    double sum = 0;
    printf("Enter an Integer: ");
    scanf("%d", &num);
    o=num;
    while (o!=0) {
        o/= 10;
        ++n;
    }
    o = num;
    while (o!= 0) {
        r= o% 10;
        sum += pow(r, n);
        o /= 10;
    }
    if ((int)sum == num)
        printf("%d is an Armstrong number.\n", num);
    else
        printf("%d is not an Armstrong number.\n", num);
    return 0;
}
