                               Fibonacci using a do while loop.



#include <stdio.h>

int main() {
    int n, i=-1;
    int a = 0, b = 1, next;

    printf("Enter the number of terms: ");
    scanf("%d", &n);

    printf("Fibonacci Series: ");

    do {
        i++;
        if (i <= 1)
            next = i;
        else {
            next = a + b;
            a = b;
            b = next;
        }
        printf("%d ", next);
    }while(i<n);

    return 0;
}

output

Enter the number of terms: 7
Fibonacci Series: 0 1 1 2 3 5 8 13
