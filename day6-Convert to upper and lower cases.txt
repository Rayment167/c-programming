                              Convert to upper and lower cases
#include <stdio.h>
#include <ctype.h>
int main() {
    char str1[100];
    int isLower = 0;
    printf("Enter a string: ");
    scanf("%s", str1); 
    if (str1[0] >= 'a' && str1[0] <= 'z') {
        isLower = 1;
    }
    if (isLower) {
        printf("To uppercase: ");
        for (int i = 0; str1[i] != '\0'; i++) {
            printf("%c", toupper(str1[i]));
        }
    } else {
        printf("To lowercase: ");
        for (int i = 0; str1[i] != '\0'; i++) {
            printf("%c", tolower(str1[i]));
        }
    }
    printf("\n");
    return 0;
}

output

Enter a string: TOLOWER
To lowercase: tolower

Enter a string: toupper
To uppercase: TOUPPER
