                                            Count vowels and consonants


#include <stdio.h>
#include <ctype.h>
void main() {
    char str[100];
    int v = 0, c = 0;
    printf("Enter a String: ");
    scanf("%s", str);
    for (int i = 0; str[i] != '\0'; i++) {
        char ch = tolower(str[i]);
        if (ch >= 'a' && ch <= 'z') {
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                v++;
            }
            else {
                c++;
            }
        }
    }
    printf("vowels = %d\nconsonants = %d\n", v, c);
}

output

Enter a String: DhivahAr
vowels = 3
consonants = 5
