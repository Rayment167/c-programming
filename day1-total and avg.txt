#include <stdio.h>
int a,b,c,d;
float total,avg;
int main() {
    printf("Enter the number a:");
    scanf("%d",&a);
    printf("Enter the number b:");
    scanf("%d",&b);
    printf("Enter the number c:");
    scanf("%d",&c);
    printf("Enter the number d:");
    scanf("%d",&d);
    total=(a+b+c+d);
    avg=(a+b+c+d)/4;
    printf("Total:%.2f\n",total);
    printf("Avg:%.2f\n",avg);

    return 0;
}
