                            to find perimeter of circle and triangle

#include <stdio.h>
int main() {
    double radius, perimeter,PI=3.14;
    printf("Enter the radius of the circle: ");
    scanf("%lf", &radius);
    perimeter = 2 * PI * radius;
    printf("Perimeter of the circle = %.2lf\n", perimeter);
    return 0;
}


output

Enter the radius of the circle: 7
Perimeter of the circle = 43.96

#include <stdio.h>
int main() {
    double s1, s2, s3, perimeter;
    printf("Enter the three sides of the triangle: ");
    scanf("%lf %lf %lf", &s1, &s2, &s3);
    perimeter = s1 + s2 + s3;
    printf("Perimeter of the triangle = %.lf\n", perimeter);
    return 0;
}

output

Enter the three sides of the triangle: 4 5 6
Perimeter of the triangle = 15
