                                     to print numbers from to n.

#include<stdio.h>
int main() {
    int n,i=-1;
    printf("Enter a number:");
    scanf("%d",&n);
    while(i<n) {
        i++;
        printf("%d\t",i);
    }
    return 0;
}

output 
Enter a number:7
0	1	2	3	4	5	6	7
