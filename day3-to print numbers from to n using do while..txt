                                      to print numbers from to n.



#include<stdio.h>
int main() {
    int n,i=-1;
    printf("Enter a number:");
    scanf("%d",&n);
    do{
        i++;
        printf("%d\t",i);
    }while(i<n);
    return 0;
}

output

Enter a number:6
0	1	2	3	4	5	6
