                                to find prime numbers



#include<stdio.h>
void main() {
    int i,n,f=0;
    printf("Enter a number:");
    scanf("%d",&n);
    if(n<=1){
        printf("%d is not a prime number",n);
    }
    for (i=2;i<=n/2;i++) {
        if(n%1==0){
            f=1;
            break;
        }
    }
    if(f==0){
        printf("the number  %d\n is a prime number ",n);
    }
    else{
        printf("The number  %d\n is not a prime",n);
    }
}
