                             Reverse order  in array

#include <stdio.h>
int main() {
    int arr[100],n,i,max;
    printf("Enter a number of elements:");
    scanf("%d",&n);
    printf("Enter %d elements:\n",n);
    for(i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    printf("array in reverse order:\n");
    for(i=n-1;i>=0;i--){
        printf("%d\t",arr[i]);
    }
    printf("\n");
    return 0;
}


output

Enter a number of elements:3
Enter 3 elements:
1 2 3
array in reverse order:
