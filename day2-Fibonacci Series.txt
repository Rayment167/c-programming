#include <stdio.h>
void main() {
    int n,f=0, s=1,next,i;
    printf("Enter the number of terms: ");
    scanf("%d", &n);
    printf("Fibonacci Series: ");
    for (i = 0; i < n; i++) {
        if (i <= 1)
            next=i;
        else {
            next = f+s;
            f=s;
            s=next;
        }
        printf("%d ", next);
    }
    printf("\t");
}


output

Fibonacci Series: 0 1 1 2 3
