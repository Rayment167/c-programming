                                   Reverse of the string

#include <stdio.h>
#include <string.h>
int main() {
    char str[100], rev[100];
    int len, i, flag = 0;
    printf("Enter a string: ");
    scanf("%s", str);
    len = strlen(str);
    for (i = 0; i < len; i++) {
        rev[i] = str[len - i - 1];
    }
    rev[len]='\0';
    printf("Reverse of the string is %s\n",rev);
    return 0;
}
