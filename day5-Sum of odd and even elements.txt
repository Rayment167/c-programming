                                      Sum of odd and even elements

#include <stdio.h>
int main() {
    int n, i;
    int even = 0,odd = 0;
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        if (arr[i] % 2 == 0)
            even += arr[i];
        else
            odd += arr[i];
    }
    printf("Sum of even elements: %d\n", even);
    printf("Sum of odd elements: %d\n", odd);
    return 0;
}
output

Enter the number of elements: 6
Enter 6 elements:
1 3 4 2 7 8
Sum of even elements: 14
Sum of odd elements: 11
